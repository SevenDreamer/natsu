       path TEXT NOT NULL UNIQUE,

       updated_at INTEGER NOT NULL
   );

       content='notes',

       INSERT INTO notes_fts(rowid, id, title, content)
       VALUES (new.rowid, new.id, new.title, '');
   END;

   CREATE TRIGGER IF NOT EXISTS notes_ad AFTER DELETE ON notes BEGIN
       INSERT INTO notes_fts(notes_fts, rowid, id, title, content)
       VALUES ('delete', old.rowid, old.id, old.title, '');
   END;

   ```rust
   use tauri::State;
   use std::sync::Mutex;
   use crate::db::Database;
   use crate::models::{Note, CreateNoteRequest, SaveNoteRequest};
   use uuid::Uuid;
   use std::fs;
   use std::path::PathBuf;

   #[tauri::command]
   pub async fn create_note(
       title: String,
       storage_path: String,
       db: State<'_, Mutex<Database>>,
   ) -> Result<Note, String> {
       // Generate UUID for note ID
       let id = Uuid::new_v4().to_string();
       let now = chrono::Utc::now().timestamp();

       // Sanitize title for filename (replace special chars with -)
       let filename = sanitize_filename(&title);
       let path = format!("wiki/{}.md", filename);
       let full_path = PathBuf::from(&storage_path).join(&path);

       // Create empty markdown file
       if let Some(parent) = full_path.parent() {
           fs::create_dir_all(parent).map_err(|e| e.to_string())?;
       }
       fs::write(&full_path, "").map_err(|e| e.to_string())?;

       let note = Note {
           id: id.clone(),
           title: title.clone(),
           content: String::new(),
           path: path.clone(),
           created_at: now,
           updated_at: now,
       };

       // Insert into database
       let conn = db.lock().unwrap();
       conn.execute(
           "INSERT INTO notes (id, title, path, created_at, updated_at) VALUES (?1, ?2, ?3, ?4, ?5)",
           rusqlite::params![&note.id, &note.title, &note.path, note.created_at, note.updated_at],
       ).map_err(|e| e.to_string())?;

       Ok(note)
   }

   #[tauri::command]
   pub async fn get_note(
       id: String,
       storage_path: String,
       db: State<'_, Mutex<Database>>,
   ) -> Result<Note, String> {
       let conn = db.lock().unwrap();
       let path: String = conn.query_row(
           "SELECT path FROM notes WHERE id = ?1",
           rusqlite::params![&id],
           |row| row.get(0),
       ).map_err(|e| e.to_string())?;

       let full_path = PathBuf::from(&storage_path).join(&path);
       let content = fs::read_to_string(&full_path).map_err(|e| e.to_string())?;

       let note = conn.query_row(
           "SELECT id, title, path, created_at, updated_at FROM notes WHERE id = ?1",
           rusqlite::params![&id],
           |row| Ok(Note {
               id: row.get(0)?,
               title: row.get(1)?,
               path: row.get(2)?,
               content: content.clone(),
               created_at: row.get(3)?,
               updated_at: row.get(4)?,
           }),
       ).map_err(|e| e.to_string())?;

       Ok(note)
   }

   #[tauri::command]
   pub async fn save_note(
       id: String,
       content: String,
       storage_path: String,
       db: State<'_, Mutex<Database>>,
   ) -> Result<(), String> {
       let now = chrono::Utc::now().timestamp();

       // Get note path from database
       let conn = db.lock().unwrap();
       let path: String = conn.query_row(
           "SELECT path FROM notes WHERE id = ?1",
           rusqlite::params![&id],
           |row| row.get(0),
       ).map_err(|e| e.to_string())?;

       // Write content to file
       let full_path = PathBuf::from(&storage_path).join(&path);
       fs::write(&full_path, &content).map_err(|e| e.to_string())?;

       // Update FTS index (for KNOW-03)
       conn.execute(
           "UPDATE notes_fts SET content = ?1 WHERE id = ?2",
           rusqlite::params![&content, &id],
       ).map_err(|e| e.to_string())?;

       // Update timestamp
       conn.execute(
           "UPDATE notes SET updated_at = ?1 WHERE id = ?2",
           rusqlite::params![now, &id],
       ).map_err(|e| e.to_string())?;

       Ok(())
   }

   #[tauri::command]
   pub async fn list_notes(
       db: State<'_, Mutex<Database>>,
   ) -> Result<Vec<Note>, String> {
       let conn = db.lock().unwrap();
       let mut stmt = conn.prepare(
           "SELECT id, title, path, created_at, updated_at FROM notes ORDER BY updated_at DESC"
       ).map_err(|e| e.to_string())?;

       let notes = stmt.query_map([], |row| {
           Ok(Note {
               id: row.get(0)?,
               title: row.get(1)?,
               path: row.get(2)?,
               content: String::new(), // Don't load content for list
               created_at: row.get(3)?,
               updated_at: row.get(4)?,
           })
       }).map_err(|e| e.to_string())?
       .collect::<Result<Vec<_>, _>>().map_err(|e| e.to_string())?;

       Ok(notes)
   }

   #[tauri::command]
   pub async fn delete_note(
       id: String,
       storage_path: String,
       db: State<'_, Mutex<Database>>,
   ) -> Result<(), String> {
       let conn = db.lock().unwrap();

       // Get note path
       let path: String = conn.query_row(
           "SELECT path FROM notes WHERE id = ?1",
           rusqlite::params![&id],
           |row| row.get(0),
       ).map_err(|e| e.to_string())?;

       // Delete file
       let full_path = PathBuf::from(&storage_path).join(&path);
       fs::remove_file(&full_path).map_err(|e| e.to_string())?;

       // Delete from database (FTS trigger handles cleanup)
       conn.execute(
           "DELETE FROM notes WHERE id = ?1",
           rusqlite::params![&id],
       ).map_err(|e| e.to_string())?;

       Ok(())
   }

   fn sanitize_filename(title: &str) -> String {
       title
           .chars()
           .map(|c| if c.is_alphanumeric() || c == '-' || c == '_' { c } else { '-' })
           .collect::<String>()
           .trim_matches('-')
           .to_lowercase()
   }
   ```

   export interface Note {
     id: string;
     title: string;
     content: string;
     path: string;
     created_at: number;
     updated_at: number;
   }

   export const notesApi = {
     create: (title: string, storagePath: string): Promise<Note> =>
       invoke('create_note', { title, storagePath }),

     get: (id: string, storagePath: string): Promise<Note> =>
       invoke('get_note', { id, storagePath }),

     save: (id: string, content: string, storagePath: string): Promise<void> =>
       invoke('save_note', { id, content, storagePath }),

     list: (): Promise<Note[]> =>
       invoke('list_notes'),

     delete: (id: string, storagePath: string): Promise<void> =>
       invoke('delete_note', { id, storagePath }),
   };

   export const storageApi = {
     selectPath: (): Promise<string | null> =>
       invoke('select_storage_path'),

     init: (path: string): Promise<void> =>

     getPath: (): Promise<string | null> =>

2. Create `src/stores/noteStore.ts`:
   ```typescript
   import { create } from 'zustand';
   import { notesApi, Note } from '../lib/tauri';

   interface NoteState {
     notes: Note[];
     activeNoteId: string | null;
     activeNote: Note | null;
     isLoading: boolean;
     error: string | null;

     loadNotes: () => Promise<void>;
     createNote: (title: string) => Promise<Note | null>;
     openNote: (id: string, storagePath: string) => Promise<void>;
     saveNote: (content: string, storagePath: string) => Promise<void>;
     deleteNote: (id: string, storagePath: string) => Promise<void>;
     setActiveNote: (note: Note | null) => void;
   }

   // Get storage path from settings store (will be created)
   const getStoragePath = () => {
     // TODO: Wire to settings store in PLAN-05
     return localStorage.getItem('storagePath');
   };

   export const useNoteStore = create<NoteState>((set, get) => ({
     notes: [],
     activeNoteId: null,
     activeNote: null,
     isLoading: false,
     error: null,

     loadNotes: async () => {
       set({ isLoading: true, error: null });
       try {
         const notes = await notesApi.list();
         set({ notes, isLoading: false });
       } catch (error) {
         set({ error: String(error), isLoading: false });
       }
     },

     createNote: async (title: string) => {
       const storagePath = getStoragePath();
       if (!storagePath) {
         set({ error: 'No storage path configured' });
         return null;
       }
       try {
         const note = await notesApi.create(title, storagePath);
         set(state => ({ notes: [note, ...state.notes] }));
         return note;
       } catch (error) {
         set({ error: String(error) });
         return null;
       }
     },

     openNote: async (id: string, storagePath: string) => {
       set({ isLoading: true });
       try {
         const note = await notesApi.get(id, storagePath);
         set({ activeNoteId: id, activeNote: note, isLoading: false });
       } catch (error) {
         set({ error: String(error), isLoading: false });
       }
     },

     saveNote: async (content: string, storagePath: string) => {
       const { activeNoteId } = get();
       if (!activeNoteId) return;

       try {
         await notesApi.save(activeNoteId, content, storagePath);
         set(state => ({
           activeNote: state.activeNote
             ? { ...state.activeNote, content, updated_at: Date.now() }
             : null,
           notes: state.notes.map(n =>
             n.id === activeNoteId ? { ...n, updated_at: Date.now() } : n
           ),
         }));
       } catch (error) {
         set({ error: String(error) });
       }
     },

     deleteNote: async (id: string, storagePath: string) => {
       try {
         await notesApi.delete(id, storagePath);
         set(state => ({
           notes: state.notes.filter(n => n.id !== id),
           activeNoteId: state.activeNoteId === id ? null : state.activeNoteId,
           activeNote: state.activeNoteId === id ? null : state.activeNote,
         }));
       } catch (error) {
         set({ error: String(error) });
       }
     },

     setActiveNote: (note: Note | null) => {
       set({ activeNote: note, activeNoteId: note?.id ?? null });
     },
   }));
   ```
